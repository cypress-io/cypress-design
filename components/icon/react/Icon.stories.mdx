import { useState } from 'react';
import Select from 'react-select';
import { ArgsTable, Canvas, Meta, Story } from '@storybook/addon-docs'
import { iconsMetadata } from '@cypress-design/icon-registry';

import { Icon } from './Icon'

# Icon

<Meta
  title="Icon"
  component={Icon}
/>

<Canvas>
  <Story name="Icon">
    {() => {
        const colors = ['blue', 'jade', 'red', 'indigo', 'purple', 'gray', 'black'];
        const [fillColor, setFillColor] = useState({label:'jade', value:'jade'});
        const [strokeColor, setStrokeColor] = useState({label:'jade', value:'jade'});
        const colorOptions = colors.map(name => ({
                value: name,
                label: name,
              }));
        return (<>
          <div className="flex items-center border-b-3 border-gray-300 pb-3">
            strokeColor
            <Select 
              defaultValue={strokeColor} 
              className="ml-2 mr-4" 
              onChange={setStrokeColor} 
              options={colorOptions} 
            />
            fillColor
            <Select
              defaultValue={fillColor} 
              className="mx-2" 
              onChange={setFillColor} 
              options={colorOptions} />
          </div>
          <div>
            {Object.keys(iconsMetadata).map((iconName) => {
              return <div key={iconName} className="flex align-center items-end border-b border-gray-300 pb-2 pt-4">
                <div className="w-45 text-right mr-4">{iconName}</div>
                {iconsMetadata[iconName].availableSizes.map((size) => {
                  return <div className="mr-4 text-center pb-1 align-bottom" key={`${iconName}_${size}`}>
                    <Icon name={iconName} size={size} strokeColor={`${strokeColor.value}-500`} fillColor={`${fillColor.value}-100`} className="m-2"/>
                    <div className="text-xs text-gray-600">{size}</div>
                  </div>
                })}
              </div>
            })}
          </div>
        </>)
    }}
  </Story>
</Canvas>

## Usage

The simple way using the Icon component

```tsx
import { Icon } from '@cypress-design/react-icon'

export const MyButtonWithIcon = () => {
  return (<button>
    <Icon name="book" size="16" strokeColor="blue-600" fillColor="red-200" className="bg-red-100" />Read
  </button>)
}
```

The tree-shakable way (more optimized)

```tsx
import { IconBook } from '@cypress-design/react-icon'

export const MyButtonWithIcon = () => {
  return (<button>
    <IconBook size="16" strokeColor="blue-600" fillColor="red-200" />
  </button>)
}
```
