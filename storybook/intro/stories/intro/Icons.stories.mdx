import { useState } from 'react';
import Select from 'react-select';
import { ArgsTable, Canvas, Meta, Story } from '@storybook/addon-docs'
import { iconsMetadata } from '@cypress-design/icon-registry';

import Icon from '@cypress-design/react-icon'

# Icons

<Meta
  title="Intro/Icons"
  component={Icon}
/>

<Canvas>
  <Story name="Icons">
    {() => {
        const colors = ['blue', 'jade', 'red', 'indigo', 'purple', 'gray', 'black'];
        const [fillColor, setFillColor] = useState({label:'jade', value:'jade'});
        const [strokeColor, setStrokeColor] = useState({label:'jade', value:'jade'});
        const [secondaryFillColor, setSecondaryFillColor] = useState({label:'indigo', value:'indigo'});
        const [secondaryStrokeColor, setSecondaryStrokeColor] = useState({label:'indigo', value:'indigo'});
        const colorOptions = colors.map(name => ({
                value: name,
                label: name,
              }));
        return (<>
          <div className="flex items-center border-b-3 border-gray-300 pb-3">
            strokeColor
            <Select 
              defaultValue={strokeColor} 
              className="ml-2 mr-4" 
              onChange={setStrokeColor} 
              options={colorOptions} 
            />
            fillColor
            <Select
              defaultValue={fillColor} 
              className="mx-2 mr-4" 
              onChange={setFillColor} 
              options={colorOptions} />
            secondaryStrokeColor
            <Select
              defaultValue={secondaryStrokeColor} 
              className="mx-2 mr-4" 
              onChange={setSecondaryStrokeColor} 
              options={colorOptions} />
            secondaryFillColor
            <Select
              defaultValue={secondaryFillColor} 
              className="mx-2" 
              onChange={setSecondaryFillColor} 
              options={colorOptions} />
          </div>
          <div>
            {Object.keys(iconsMetadata).map((iconName) => {
              const meta = iconsMetadata[iconName]
              return <div key={iconName} className="flex align-center items-end border-b border-gray-300 pb-2 pt-4">
                <div className="w-60 text-right mr-4">{iconName}</div>
                {meta.availableSizes.map((size) => {
                  return (<div key={`${iconName}_${size}`} className="mr-8 flex">
                    <div className="mr-2 text-center pb-1 align-bottom">
                      <Icon 
                        name={iconName} 
                        size={size} 
                        strokeColor={`${strokeColor.value}-400`} 
                        fillColor={`${fillColor.value}-50`} 
                        secondaryStrokeColor={`${secondaryStrokeColor.value}-400`}
                        secondaryFillColor={`${secondaryFillColor.value}-50`}
                        className="m-2"/>
                      <div className="text-xs text-gray-600">{size}</div>
                    </div>
                    <div key={`${iconName}_${size}_colors`} className="text-center">
                      {meta.hasStrokeColor && meta.hasStrokeColor.includes(size) && <div className={`px-1 my-1 text-xs text-${strokeColor.value}-500`}>s</div>}
                      {meta.hasFillColor && meta.hasFillColor.includes(size) && <div className={`px-1 my-1 text-xs text-gray-800 bg-${fillColor.value}-50`}>f</div>}
                      {meta.hasSecondaryStrokeColor && meta.hasSecondaryStrokeColor.includes(size) && <div className={`px-1 my-1 text-xs text-${secondaryStrokeColor.value}-500`}>ss</div>}
                      {meta.hasSecondaryFillColor && meta.hasSecondaryFillColor.includes(size) && <div className={`px-1 my-1 text-xs text-gray-800 bg-${secondaryFillColor.value}-50`}>sf</div>}
                    </div>
                  </div>)
                })}
              </div>
            })}
          </div>
        </>)
    }}
  </Story>
</Canvas>
